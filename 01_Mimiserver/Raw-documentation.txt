#!/bin/bash

# Flush existing rules
iptables -F
iptables -t nat -F
iptables -t mangle -F
iptables -X

#INPUT
iptables -A INPUT -i wg0 -s 10.7.7.7/32 -p udp --dport 53 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.7/32 -p tcp --dport 53 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.7/32 -p tcp --dport 22 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.7/32 -p tcp --dport 443 -j ACCEPT

iptables -A INPUT -i wg0 -s 10.7.7.4/32 -p udp --dport 53 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.4/32 -p tcp --dport 53 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.4/32 -p tcp --dport 22 -j ACCEPT
iptables -A INPUT -i wg0 -s 10.7.7.4/32 -p tcp --dport 443 -j ACCEPT

iptables -A INPUT -i lo -j ACCEPT
iptables -A INPUT -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT
iptables -A INPUT -i eth0 -p udp --dport 8443 -j ACCEPT

iptables -A INPUT -i eth0 -s 192.168.1.0/24 -j DROP
iptables -A INPUT -i wg0 -s 192.168.1.0/24 -j DROP
iptables -A INPUT -j DROP

# FORWARD
iptables -A FORWARD -i wg0 -o eth0 -s 10.7.7.4/32 -j ACCEPT
iptables -A FORWARD -i wg0 -o eth0 -s 10.7.7.7/32 -j ACCEPT
iptables -A FORWARD -i eth0 -o wg0 -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT
iptables -A FORWARD -i eth0 -o wg0 -j DROP
iptables -A FORWARD -i eth0 -s 192.168.1.0/24 -j DROP
iptables -A FORWARD -j LOG --log-prefix "FORWARD DROP: "
iptables -A FORWARD -j DROP

# OUTPUT
iptables -A OUTPUT -o lo -j ACCEPT
iptables -A OUTPUT -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT 
iptables -A OUTPUT -o wg0 -d 192.168.1.0/24 -j DROP
iptables -A OUTPUT -o wg0 -j ACCEPT
iptables -A OUTPUT -o eth0 -d 192.168.1.0/24 -j DROP
iptables -A OUTPUT -o eth0 -j ACCEPT
iptables -A OUTPUT -j DROP

#NAT
iptables -t nat -A POSTROUTING -o eth0 -s 10.7.7.4/32 -j MASQUERADE
iptables -t nat -A POSTROUTING -o eth0 -s 10.7.7.7/32 -j MASQUERADE

#iptables -A INPUT -i eth0 -s 192.168.1.0/24 -j LOG --log-prefix "LAN BLOCK INPUT: "
#iptables -A INPUT -i wg0 -s 192.168.1.0/24 -j LOG --log-prefix "LAN BLOCK WG INPUT: "

#make it executable
chmod +x ~/iptables-mimiserver.sh

#set default drop policy
iptables -P INPUT DROP
iptables -P FORWARD DROP
iptables -P OUTPUT DROP

#DISABLE UFW
sudo ufw disable
sudo ufw status
