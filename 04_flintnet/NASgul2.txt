OMV was buggy and annoying. plugins were nonexistent, and services added to the os separately conflicted with OMV. I decided it would feel cleaner to me if I manually configured samba to provide NAS in order to free up flexibility with the os. In this way I could install docker along side samba and still have NAS functionality while i tinker with containers and work my way to up to trying macvlans. Ill first rehearse a docker stack with pihole, unbound, vaultwarde, Prometheus, Grafana, suricata, and kuma on a user bridge network. then if it works ill deploy it on the pi that has smb working so far.

Step 1: install docker
+++++++++
#install dependencies
sudo apt update
sudo apt install ca-certificates curl gnupg
#add dockers official gpg key
sudo install -m 0755 -d /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
sudo chmod a+r /etc/apt/keyrings/docker.gpg
#add docker repo, first find os version codename
cat /etc/os-release #find version codename and add to the end of URL
#ubuntu jammy
echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/<ADD VERSION CODENAME HERE> stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
#update new repository and install 
sudo apt update
sudo apt install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
# enable on boot
sudo systemctl enable docker
sudo systemctl start docker
# add user to docker group to avoid having to type sudo
sudo usermod -aG docker <username>

STEP5 vaultwarden and self-signed rootCA
#create folder
mkdir -p ~/myCA
cd ~/myCA
#create private key
openssl genrsa -out myCA.key 4096
#create root certificate
openssl req -x509 -new -nodes -key myCA.key -sha256 -days 3650 -out myCA.pem \
  -subj "/CN=Banana-Root-CA"
#this creates myCA.key and myCA.pem
#create vaultwarden certificate
#create the private key
openssl genrsa -out vaultwarden.lan.key 4096
#create the csr
openssl req -new -key vaultwarden.lan.key -out vaultwarden.lan.csr \
  -subj "/CN=vaultwarden.lan"
#sign it with CA
openssl x509 -req -in vaultwarden.lan.csr \
  -CA myCA.pem -CAkey myCA.key -CAcreateserial \
  -out vaultwarden.lan.crt -days 825 -sha256
#place certs
cd /srv/docker/vaultwarden
mkdir ssl
cp ~/myCA/vaultwarden.lan.crt ssl/
cp ~/myCA/vaultwarden.lan.key ssl/
cp ~/myCA/myCA.pem ssl/
cd /srv/docker/vaultwarden
sudo vi docker-compose.yml
services:
  vaultwarden:
    image: vaultwarden/server:latest
    container_name: vaultwarden
    restart: unless-stopped
    networks:
      - electopia
    ports:
      - "8081:80"
    volumes:
      - ./data:/data
      - ./ssl:/data/ssl
    environment:
      WEBSOCKET_ENABLED: "true"
      SIGNUPS_ALLOWED: "false"
      DOMAIN: "https://vaultwarden.lan"
      ROCKET_TLS: '{certs="/data/ssl/vault.lan.crt",key="/data/ssl/vault.lan.key"}'
      ALLOW_HTTP: "true"

networks:
  electopia:
    external: true
docker compose up -d
docker ps
