#create folder
mkdir -p ~/myCA
cd ~/myCA
#create private key
openssl genrsa -out myCA.key 4096
#create root certificate
openssl req -x509 -new -nodes -key myCA.key -sha256 -days 3650 -out myCA.pem \
  -subj "/CN=Banana-Root-CA"
#this creates myCA.key and myCA.pem
#create vaultwarden certificate
#create the private key
openssl genrsa -out vaultwarden.lan.key 4096
#create the csr
openssl req -new -key vaultwarden.lan.key -out vaultwarden.lan.csr \
  -subj "/CN=vaultwarden.lan"
#sign it with CA
openssl x509 -req -in vaultwarden.lan.csr \
  -CA myCA.pem -CAkey myCA.key -CAcreateserial \
  -out vaultwarden.lan.crt -days 825 -sha256

# go to nasgul cd ~/myCA
#create SAN config file
sudo vi wildcard-hub-openssl.cnf
# put this inside it ensures wildcard works, bare domain hub works, npm can terminate https for any *.hub service
[ req ]
default_bits       = 2048
prompt             = no
default_md         = sha256
req_extensions     = req_ext
distinguished_name = dn

[ dn ]
CN = *.hub

[ req_ext ]
subjectAltName = @alt_names

[ alt_names ]
DNS.1 = *.hub
DNS.2 = hub
#generate private key
openssl genrsa -out wildcard.hub.key 2048
#generate the csr
openssl req -new -key wildcard.hub.key \
-out wildcard.hub.csr \
-config wildcard-hub-openssl.cnf
# sign it with Banana-root-CA
openssl x509 -req \
-in wildcard.hub.csr \
-CA myCA.pem \
-CAkey myCA.key \
-CAcreateserial \
-out wildcard.hub.crt \
-days 825 \
-sha256 \
-extfile wildcard-hub-openssl.cnf \
-extensions req_ext
