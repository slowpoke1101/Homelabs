        iif "wg0" ip saddr 10.7.7.4 tcp dport 22 accept
        iif "wg0" ip saddr 10.7.7.4 tcp dport 443 accept

        # Loopback
        iif "lo" accept

        # Established connections
        ct state established,related accept

        # Allow UDP 8443 on eth0
        iif "eth0" udp dport 8443 accept

        # LAN isolation
        iif "eth0" ip saddr 192.168.1.0/24 drop
        iif "wg0" ip saddr 192.168.1.0/24 drop
    }

    chain forward {
        type filter hook forward priority 0;
        policy drop;

        # Allow WG clients to forward to eth0
        iif "wg0" ip saddr 10.7.7.4 oif "eth0" accept
        iif "wg0" ip saddr 10.7.7.7 oif "eth0" accept

        # Block LAN
        iif "eth0" ip saddr 192.168.1.0/24 drop
    }

        type filter hook output priority 0;
